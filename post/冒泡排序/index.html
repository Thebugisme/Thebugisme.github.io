冒泡排序c代码 

#include <stdio.h>
int main()
{
    int swap,len,i,j;
    scanf("%d",&len);
    int a[len];
        for(i=0;i<len;i++)
        {
            scanf("%d",&a[i]);for(i=0;i<len-1;i++);
            for(j=0;j<len-1;j++)
            {
                if(a[j]>a[j+1])
                {
                    swap=a[j];
                    a[j]=a[j+1];
                    a[j+1]=swap;
                }   
            }
        }
        for(i=0;i<len;i++)
        {
            printf("%d",a[i]);
        }
}
