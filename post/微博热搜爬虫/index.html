import requests
from bs4 import BeautifulSoup
import  os
import csv

headers = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36',
    'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,*/*;q=0.8',
    'Accept-Language': 'zh-CN,zh;q=0.9,en;q=0.8',
    'Accept-Encoding': 'gzip, deflate, br',
    'Connection': 'keep-alive',
    'Upgrade-Insecure-Requests': '1'
}

cookies = {
    'SUB': 'SUB=_2AkMetCZrf8NxqwFRm_sdyGnkZIlzyQnEieKo6NewJRMxHRl-yT9yqmwEtRB6NTQIhM8PgNZxkRl0UjRDlXVtlPI_DGu7; SUBP=0033WrSXqPxfM72-Ws9jqgMF55529P9D9WW0UmaW8EXux3FjJLnaV06_; _s_tentry=cn.bing.com; UOR=cn.bing.com,s.weibo.com,cn.bing.com; Apache=1677554318656.811.1776855460946; SINAGLOBAL=1677554318656.811.1776855460946; ULV=1776855460958:1:1:1:1677554318656.811.1776855460946:',  # 需要替换成你的实际cookie值
    # 可以添加更多cookie键值对
}

url = "https://s.weibo.com/top/summary/"
r = requests.get(url, headers=headers, cookies=cookies)
soup = BeautifulSoup(r.text, 'html.parser')
items = soup.find_all('td', class_='td-02')
# 准备数据并写入CSV
desktop = os.path.join(os.path.expanduser('~'), 'Desktop')
file_path = os.path.join(desktop, 'weibo_hotsearch.csv')

#保存数据到excel
with open(file_path, 'w', newline='', encoding='utf-8-sig') as csvfile:
    writer = csv.writer(csvfile)
    writer.writerow(['标题', '热度'])

    for item in items:
        title = item.find('a').text
        rating = item.find('span').text if item.find('span') else ''
        print(title, rating)
        writer.writerow([title, rating])

print(f"数据已保存到：{file_path}")
